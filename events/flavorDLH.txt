# The Marathi Revolt
country_event = {
	id = flavor_dlh.1
	title = "flavor_dlh.EVTNAME1"
	desc = "flavor_dlh.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		ai = no
		is_year = 1625
		NOT = { exists = MAR }
		OR = {
			NOT = { religion_group = dharmic }
			NOT = { accepted_culture = marathi }
		}
		is_subject = no
		num_of_cities = 5
		any_owned_province = {
			culture = marathi
			religion_group = dharmic
		}
	}
	
	mean_time_to_happen = {
		months = 2000
		modifier = {
			factor = 0.75
			NOT = { accepted_culture = marathi }
		}
		modifier = {
			factor = 0.75
			NOT = { primary_culture = marathi }
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 2
				culture = marathi
				religion_group = dharmic
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 4
				culture = marathi
				religion_group = dharmic
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 6
				culture = marathi
				religion_group = dharmic
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 8
				culture = marathi
				religion_group = dharmic
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					religion_group = dharmic
					culture = marathi
				}
				discover_country = MAR
				add_core = MAR
			}
			release = MAR
			every_province = {
				limit = {
					culture = marathi
				}
				add_claim = MAR
			}
		}
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA1"
		MAR = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
		hidden_effect = {
			MAR = {
				country_event = { id = flavor_mar.1 days = 1 }
			}
		}
	}
}

# The Lodi Dynasty
country_event = {
	id = flavor_dlh.3
	title = "flavor_dlh.EVTNAME3"
	desc = "flavor_dlh.EVTDESC3"
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = DLH
		is_year = 1452
		NOT = { is_year = 1553 }
		dynasty = "Lodi" 
		OR = { 
			NOT = { dip = 0 } 
			NOT = { mil = 0 } 
			NOT = { adm = 0 }
			NOT = { legitimacy = 70 }
			NOT = { stability = 1 }
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA3"
		add_stability = -1
		every_owned_province = {
			limit = { is_core = MAR }
			add_unrest = 2
		}
		add_inflation = 1
		set_country_flag = dlh_lodi_flag
	}
}
