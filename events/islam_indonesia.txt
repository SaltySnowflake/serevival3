namespace = indislam

# Java and Borneo

country_event = {
	id = indislam.1
	title = "indislam.EVTNAME1"
	desc = "indislam.EVTDESC1"
	picture = RELIGION_eventPicture

	trigger = {
		is_year = 1475
		NOT = { has_ruler_flag = indislamrulerb }
		NOT = { has_country_modifier = "reformed_hindu" }
		any_owned_province = {
			region = indonesia_region				
			OR = {
				culture_group = malay
				culture = moluccan
			}
			OR = {
				religion = buddhism
				religion_group = pacific_pagan
				religion = hinduism
			}
		}
	}
	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.8
			religion = sunni
		}	
	}
	immediate = {
		random_owned_province = {
			limit = {
				region = indonesia_region
				OR = {
					culture_group = malay
					culture = moluccan
				}
				OR = {
					religion = buddhism
					religion_group = pacific_pagan
					religion = hinduism
				}
			}
			set_province_flag = indoislam_flag
		}
	}
	option = {
		name = "indislam.1a"
		every_owned_province = {
			limit = { has_province_flag = indoislam_flag }
			change_religion = sunni
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 7300
			}
			clr_province_flag = indoislam_flags
		}
	}
}

					religion = buddhism

			}

		}

		name = "indislam.2b"
		set_country_flag = indislamrulerb

			limit = { religion = sunni }

# Muhammad Kebungsuwan

country_event = {
	id = indislam.8
	title = "indislam.EVTNAME8"
	desc = "indislam.EVTDESC8"
	picture = RELIGION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = MGD
		NOT = { religion = sunni }
		is_year = 1495
	}
	mean_time_to_happen = {
		months = 60
	}
	option = { # make him sultan
		name = "indislam.8a"
		ai_chance = { factor = 75 }
		add_stability = -2
		change_religion = sunni
		define_ruler = {
			name = "Muhammad"
			dynasty = "Kebungsuwan"
		}
	}
	option = { # resist him
		name = "indislam.8b"
		ai_chance = { factor = 25 }
		add_stability = -1
		spawn_rebels = {
			type = pretender_rebels
			size = 2
			leader = "Muhammad Kebungsuwan"
		}
	}
}
