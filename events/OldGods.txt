namespace = oldgods
# A law against the killing of cats
country_event = {
	id = oldgods.1
	title = "oldgods.1.t"
	desc = "oldgods.1.d"
	picture = COUNTRYSIDE_eventPicture

	trigger = {
		is_pagan_trigger = yes
		NOT = { has_ruler_flag = the_old_ones_are_waking }
		NOT = { has_country_flag = the_old_ones_are_waking }
		num_of_owned_provinces_with = {
			value = 5
			has_port = yes
		}
		has_heir = yes 
		heir_age = 15
	}
	
	mean_time_to_happen = { 
		months = 1000 
		modifier = {
			factor = 5
			NOT = { num_of_cities = 25 }
		}
		modifier = {
			factor = 0.5
			ruler_has_personality = lunatic_personality
		}
		modifier = {
			factor = 0.5
			num_of_cities = 50
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				has_port = yes
			}
		}
	}
	
	immediate = { set_ruler_flag = the_old_ones_are_waking }
	
	option = {
		name = "oldgods.1.a"
		custom_tooltip = dreamerstirs
		hidden_effect = {
			country_event = { id = oldgods.2 days = 14 }
			if = {
				limit = { NOT = { ruler_has_personality = lunatic_personality } }
				clear_scripted_personalities = yes
				add_ruler_personality = lunatic_personality
			}
		}
	}
	
	option = {
		name = "oldgods.1.b"
		add_prestige = 10
	}
}

		add_mil_power = 250
		add_prestige = 50
		add_legitimacy = 15

			limit = { NOT = { ruler_has_personality = scholar_personality } }

		}

		name = "oldgods.6.b"
		change_mil = 3

		}

			shock = 4
			fire = 4
			manuever = 4
			siege = 4
			#trait = glory_seeker_personality
		}
	}

}
