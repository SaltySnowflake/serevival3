vinland_native_trigger = {
	AND = {
		tag = VIN
		any_owned_province = {
			AND = {
				has_province_flag = vinland_nat_settlers
				is_city = yes
			}
		}
		NOT = {
			AND = {
				has_country_flag = vinland_natives
				OR = {
					check_variable = {
						which = vinland_natives
						value = 15
					}
					NOT = {
						check_variable = {
							which = vinland_natives
							value = 1
						}
					}
				}
			}
		}
	}
}

vinland_native_province = {
	AND = {
		has_province_flag = vinland_nat_settlers
		is_city = yes
		owner = { vinland_native_trigger = yes }
	}
}
