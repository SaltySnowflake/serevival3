vinland_nationalists = {
	
	color = { 25 180 25 }
	
	area = nation 					# Stays within the nation
	government = any
	defection = none					# Will defect to the "best" alternative.
	independence = none				# Does not strive for independence
	defect_delay = 120
	gfx_type = culture_province
	will_relocate = yes	
	
	resilient = no					# Disbands on retreat
	reinforcing = yes
	general = yes				
	smart = yes
	unit_transfer = no				# Units do not remain after enforcing demands.
	will_relocate = yes
	
	artillery = 0.0
	infantry = 0.9
	cavalry = 0.1
	morale = 0.8
	
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no	
	handle_action_change_culture = yes
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0
			owner = { NOT = { tag = VIN } }
		}
		modifier = {
			factor = 0
			NOT = { has_province_flag = vinland_nat_settlers }
		}
		modifier = {
			factor = 100.0
			has_province_modifier = vinland_nationalist_insurrection
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 3
			has_province_flag = vinland_nat_settlers
		}
		modifier = {
			factor = 1.5
			has_province_modifier = vinland_native_takeover
		}
		modifier = {
			factor = 1.5
			has_province_modifier = vinland_native_autonomy
		}
		modifier = {
			factor = 1.5
			has_province_modifier = vinland_native_bribe
		}
		modifier = {
			factor = 1.5
			has_province_modifier = vinland_native_insurrection
		}
		modifier = {
			factor = 1.5
			has_province_modifier = vinland_nationalist_insurrection
		}
	}
	
 	# Province Scope
 	siege_won_trigger = {
 		always = yes
 	}
	siege_won_effect = {
		if = {
			limit = { owner = { has_country_flag = vinland_event_25 } }
			add_province_modifier = {
				name = vinland_exile_natives
				duration = 3650
			}
			hidden_effect = {
				clr_province_flag = vinland_nat_settlers
				owner = { vinland_reb_segregate = yes }
				if = {
					limit = { has_province_modifier = vinland_native_autonomy }
					remove_province_modifier = vinland_native_autonomy
				}
				if = {
					limit = { has_province_modifier = vinland_native_bribe }
					remove_province_modifier = vinland_native_bribe
				}
				if = {
					limit = { has_province_modifier = vinland_native_insurrection }
					remove_province_modifier = vinland_native_insurrection
				}
				if = {
					limit = { has_province_modifier = vinland_nationalist_insurrection }
					remove_province_modifier = vinland_nationalist_insurrection
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "vinland_nationalists_demand"
	
	# Country scope
	demands_enforced_effect = {
		every_owned_province = {
			limit = {
				has_province_flag = vinland_nat_settlers
			}
			add_province_modifier = {
				name = vinland_exile_natives
				duration = 3650
			}
			hidden_effect = {
				clr_province_flag = vinland_nat_settlers
				if = {
					limit = { has_province_modifier = vinland_native_autonomy }
					remove_province_modifier = vinland_native_autonomy
				}
				if = {
					limit = { has_province_modifier = vinland_native_bribe }
					remove_province_modifier = vinland_native_bribe
				}
				if = {
					limit = { has_province_modifier = vinland_native_insurrection }
					remove_province_modifier = vinland_native_insurrection
				}
				if = {
					limit = { has_province_modifier = vinland_nationalist_insurrection }
					remove_province_modifier = vinland_nationalist_insurrection
				}
			}
		}
		add_prestige = -50
	}
}
